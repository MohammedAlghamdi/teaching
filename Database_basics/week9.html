Week 9: Indexes and Views

Employee Table


        CREATE TABLE emp (
            EMPNO NUMBER(4) PRIMARY KEY,
            ENAME VARCHAR2(10),
            JOB VARCHAR2(9),
            MGR NUMBER(4),
            HIREDATE DATE,
            SAL NUMBER(7, 2),
            COMM NUMBER(7, 2),
            DEPTNO NUMBER(2));

Department Table


        CREATE TABLE dept (
            DEPTNO NUMBER(2) PRIMARY KEY,
            DNAME VARCHAR2(14),
            LOC VARCHAR2(13));

Creating an Index

Creating a simple index 


        CREATE INDEX idx_deptno ON emp(deptno);

Creating a non-unique index 


        CREATE INDEX idx_salary ON emp(sal);

Creating a composite index


        CREATE INDEX idx_ename_job ON emp(ename, job);

Example of dropping an index 


        DROP INDEX idx_deptno;

View Creation

View Creation 


CREATE VIEW dept10_employees AS
SELECT * FROM emp
WHERE deptno = 10;


CREATE VIEW dept20_employees AS
SELECT * FROM emp
WHERE deptno = 20;


CREATE VIEW salesman_details AS
SELECT ename, job, sal FROM emp
WHERE job = 'SALESMAN';

Querying the Views 


    SELECT * FROM dept10_employees;


    SELECT * FROM dept20_employees;
    

    SELECT * FROM salesman_details;
